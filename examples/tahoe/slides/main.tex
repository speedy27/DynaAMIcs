% ---------------- 1. PREMISE ----------------
\begin{frame}{The premise --- JEPA for cell biology}
\vspace{6pt}
\begin{card}[Predict, don't reconstruct]
Single-cell transcriptomes are \hl{noisy, sparse, high-dimensional sets} of genes.
Reconstructing counts (scGPT) is wasteful; \hl{predicting in a learned latent space}
(LeCun's JEPA) discards task-irrelevant noise. \emph{GeneJEPA} (Litman 2025) proved this on Tahoe-100M.
\end{card}
\vspace{6pt}
\begin{card2}[Our goal]
A \hk{true Energy-Based JEPA world model} of drug response ---
\hk{(control cell, drug) $\rightarrow$ perturbed cell} --- on a \hk{frozen pretrained encoder},
with \hk{biology-specific losses}, beating the standard baselines.
\end{card2}
\end{frame}

% ---------------- 4. RECIPE ----------------
\begin{frame}{The EB-JEPA recipe in one equation}
\vspace{10pt}
\begin{card}[Energy $=$ prediction error in representation space]
\centering\vspace{2pt}
$\displaystyle \mathcal{L} = \big\lVert\, g_\phi(f_\theta(x),\, q_\omega(a)) - f_\theta(x')\,\big\rVert^2 \;+\; \lambda\,\mathcal{R}(z)$
\vspace{2pt}
\end{card}
\vspace{10pt}
\begin{itemize}\setlength\itemsep{5pt}
  \item \hl{$f_\theta$} encoder \quad \hl{$g_\phi$} predictor \quad \hl{$q_\omega$} action encoder \quad \hl{$\mathcal{R}$} anti-collapse regularizer
  \item Three settings: image (identity $g_\phi$) $\cdot$ video (predict next) $\cdot$ \hk{action-conditioned world model} --- ours
\end{itemize}
\end{frame}

% ---------------- 5. ARCHITECTURE ----------------
\begin{frame}{Our model --- frozen foundation encoder $+$ learned drug predictor}
\vspace{4pt}
\begin{columns}[T]
\begin{column}{0.58\textwidth}\small
\begin{itemize}\setlength\itemsep{6pt}
  \item \hl{$f_\theta$} $=$ \textbf{frozen} MosaicFM-3B embedding (2560-d), a pretrained single-cell foundation encoder
  \item \hl{$g_\phi$} $=$ GRU predictor, \hl{conditioned on the drug} (action $a$)
  \item Energy $=\lVert g_\phi(z_{\text{ctrl}}, a) - z_{\text{pert}}\rVert^2$ via \texttt{JEPA.unroll}
  \item \hk{Only the predictor is trained}; a drug is a large, controlled state change $\Rightarrow$ \hk{no collapse}
\end{itemize}
\end{column}
\begin{column}{0.40\textwidth}
\begin{card2}[Pipeline]
\footnotesize
$z_{\text{ctrl}}$ \textcolor{muted}{(DMSO cell)}\\[3pt]
$\downarrow$\; $g_\phi(\cdot,\ \text{drug})$\\[3pt]
$\hat z_{\text{pert}}$\\[3pt]
$\downarrow$\; energy vs real $z_{\text{pert}}$
\end{card2}
\end{column}
\end{columns}
\end{frame}

% ---------------- 6. LOSSES ----------------
\begin{frame}{Our contribution --- three biology-aware loss terms}
\vspace{2pt}{\footnotesize\color{muted}Track A (trainable encoder): SIGReg $+$ Pathway\quad$\cdot$\quad Track B (frozen encoder $\Rightarrow$ no anti-collapse): PerturbationSignature $+$ Pathway}\par
\vspace{5pt}\small
\begin{card}[1. SIGReg --- anti-collapse (track A; our choice vs GeneJEPA's VICReg)]
Forces the latent batch toward an \hl{isotropic Gaussian} (Epps--Pulley statistic on random 1-D projections).
One hyper-parameter, linear cost, steadier than VICReg's two coefficients.
\end{card}
\vspace{4pt}
\begin{card}[2. PathwayCoherenceLoss --- gene-program prior]
Latent pairwise distances must match \hl{co-expression-module activity} distances --- injects biological program structure.
\end{card}
\vspace{4pt}
\begin{card2}[3. PerturbationSignatureLoss --- drug consistency (track B)]
Predicted shift $\hat z_{\text{pert}}-z_{\text{ctrl}}$ points the \hk{same direction for the same drug} (a compound's signature).
\end{card2}
\end{frame}
